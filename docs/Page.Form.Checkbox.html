Module Page.Form.Checkbox

module Checkbox: sig .. end

Operation on Checkboxes

val value : Page.Form.checkbox Page.Form.input -> string

Return the value (the label) of a checkbox

val values : Page.Form.t -> Page.Form.checkbox Page.Form.input -> string list

Return the values of all the checkboxes with the same name as the given one

val choices : Page.Form.t ->
       Page.Form.checkbox Page.Form.input ->
       Page.Form.checkbox Page.Form.input Page.seq

Return all the checkboxes with the same name as the given one

val checked : Page.Form.t -> Page.Form.checkbox Page.Form.input -> string list

Return the list of all checked checkboxes with the same name as the given one

val check : Page.Form.t -> Page.Form.checkbox Page.Form.input -> Page.Form.t

check form cb return form where cb is checked

val uncheck : Page.Form.t -> Page.Form.checkbox Page.Form.input -> Page.Form.t

uncheck form cb return form where cb is unchecked

val is_checked : Page.Form.t -> Page.Form.checkbox Page.Form.input -> bool

Check if the specified checkbox is checked

val checked_default : Page.Form.t -> Page.Form.checkbox Page.Form.input -> string list

Values with the checked attribute set

val reset : Page.Form.t -> Page.Form.checkbox Page.Form.input -> Page.Form.t

Reset to its default value, meaning that only the checkboxes with the checked attribute will be checked
